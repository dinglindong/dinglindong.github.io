# еҜ№дәҺMVVMзҡ„зҗҶи§Ј

MVVM жҳҜ Model-View-ViewModel зҡ„зј©еҶҷгҖӮ

Modelд»ЈиЎЁж•°жҚ®жЁЎеһӢпјҢд№ҹеҸҜд»ҘеңЁModelдёӯе®ҡд№үж•°жҚ®дҝ®ж”№е’Ңж“ҚдҪңзҡ„дёҡеҠЎйҖ»иҫ‘гҖӮ

View д»ЈиЎЁUI з»„д»¶пјҢе®ғиҙҹиҙЈе°Ҷж•°жҚ®жЁЎеһӢиҪ¬еҢ–жҲҗUI еұ•зҺ°еҮәжқҘгҖӮ

ViewModel зӣ‘еҗ¬жЁЎеһӢж•°жҚ®зҡ„ж”№еҸҳе’ҢжҺ§еҲ¶и§ҶеӣҫиЎҢдёәгҖҒеӨ„зҗҶз”ЁжҲ·дәӨдә’пјҢз®ҖеҚ•зҗҶи§Је°ұжҳҜдёҖдёӘеҗҢжӯҘView е’Ң Modelзҡ„еҜ№иұЎпјҢиҝһжҺҘModelе’ҢViewгҖӮ

# v-ifе’Ңv-showзҡ„еҢәеҲ«

v-show д»…д»…жҺ§еҲ¶е…ғзҙ зҡ„жҳҫзӨәж–№ејҸпјҢе°ҶдҪҝз”Ёcssзҡ„displayеұһжҖ§еңЁblockе’ҢnoneжқҘеӣһеҲҮжҚў;иҖҢv-ifдјҡжҺ§еҲ¶иҝҷдёӘDOMиҠӮзӮ№жҳҜеҗҰеӯҳеңЁгҖӮ

еҪ“жҲ‘д»¬йңҖиҰҒз»ҸеёёеҲҮжҚўжҹҗдёӘе…ғзҙ зҡ„жҳҫзӨә/йҡҗи—Ҹж—¶пјҢдҪҝз”Ёv-showдјҡжӣҙеҠ иҠӮзңҒжҖ§иғҪдёҠзҡ„ејҖй”Җ;еҪ“еҸӘйңҖиҰҒдёҖж¬ЎжҳҫзӨәжҲ–иҖ…йҡҗи—Ҹж—¶пјҢдҪҝз”Ёv-ifжӣҙеҠ еҗҲзҗҶгҖӮ

# watchе’Ңcomputed

1.computedжҳҜз”ЁдәҺи®Ўз®—зҡ„пјҢwatchжҳҜз”ЁдәҺзӣ‘еҗ¬ж•°жҚ®зҡ„гҖӮ

2.computedи®Ўз®—зҡ„ж•°жҚ®еҸҜд»ҘдёҚеӯҳеңЁпјҢwatchзӣ‘еҗ¬зҡ„ж•°жҚ®еҝ…йЎ»еӯҳеңЁгҖӮ

3.computedдјҡиў«зј“еӯҳпјҢwatchдёҚдјҡиў«зј“еӯҳ

# vueз»ҷdataдёӯзҡ„еҜ№иұЎж·»еҠ дёҖдёӘиЎҢеұһжҖ§

<template>
    <div>
        <ul>
            <li v-for="value in obj" :key="value">{{value}}</li>
        </ul>
        <button @click="addObjB">ж·»еҠ obj.b</button>
    </div>
</template>

<script>
export default {
    data () {
        return {
            obj: {
                a: 'obj.a'
            }
        }
    },
    methods: {
        addObjB () {
            this.obj.b = 'obj.b'
            console.log(this.obj)
        }
    }
}
</script>

зӮ№еҮ»buttonдјҡеҸ‘зҺ°пјҢobj.b е·Із»ҸжҲҗеҠҹж·»еҠ пјҢдҪҶжҳҜи§Ҷеӣҫе№¶жңӘеҲ·ж–°пјҡ

еҺҹеӣ еңЁдәҺеңЁVueе®һдҫӢеҲӣе»әж—¶пјҢobj.bе№¶жңӘеЈ°жҳҺпјҢеӣ жӯӨе°ұжІЎжңүиў«VueиҪ¬жҚўдёәе“Қеә”ејҸзҡ„еұһжҖ§пјҢиҮӘз„¶е°ұдёҚдјҡи§ҰеҸ‘и§Ҷеӣҫзҡ„жӣҙж–°пјҢиҝҷж—¶е°ұйңҖиҰҒдҪҝз”ЁVueзҡ„е…ЁеұҖapi $set()пјҡ

addObjB () {
    // this.obj.b = 'obj.b'
    this.$set(this.obj, 'b', 'obj.b')
    console.log(this.obj)
}

$set()ж–№жі•зӣёеҪ“дәҺжүӢеҠЁзҡ„еҺ»жҠҠobj.bеӨ„зҗҶжҲҗдёҖдёӘе“Қеә”ејҸзҡ„еұһжҖ§пјҢжӯӨж—¶и§Ҷеӣҫд№ҹдјҡи·ҹзқҖж”№еҸҳдәҶ

# vueдёӯж·ұеәҰзӣ‘еҗ¬жҹҗдёӘеҖј

жҜ”еҰӮзҺ°еңЁйңҖиҰҒзӣ‘жҺ§dataдёӯпјҢobj.a зҡ„еҸҳеҢ–гҖӮVueдёӯзӣ‘жҺ§еҜ№иұЎеұһжҖ§зҡ„еҸҳеҢ–дҪ еҸҜд»Ҙиҝҷж ·пјҡ

watch: {
    obj: {
        handler(newValue,oldValue){
            console.log('obj changed')
        },
        deep: true
    }
}

deepеұһжҖ§иЎЁзӨәж·ұеұӮйҒҚеҺҶпјҢдҪҶжҳҜиҝҷд№ҲеҶҷдјҡзӣ‘жҺ§objзҡ„жүҖжңүеұһжҖ§еҸҳеҢ–пјҢе№¶дёҚжҳҜжҲ‘д»¬жғіиҰҒзҡ„ж•ҲжһңпјҢжүҖд»ҘеҒҡзӮ№дҝ®ж”№пјҡ

watch: {
    'obj.a': {
        handler(newName,oldName){
            console.log('obj.a changed')
        }
    }
}

иҝҳжңүдёҖз§Қж–№жі•пјҢеҸҜд»ҘйҖҡиҝҮcomputedжқҘе®һзҺ°пјҢеҸӘйңҖиҰҒ

computed: {
    a1(){
        return this.obj.a
    }
}

иҝҷз§Қж–№ејҸеҲ©з”Ёи®Ўз®—еұһжҖ§зҡ„зү№жҖ§жқҘе®һзҺ°пјҢеҪ“дҫқиө–ж”№еҸҳж—¶пјҢдҫҝдјҡйҮҚж–°и®Ўз®—дёҖдёӘж–°еҖјгҖӮ

# vueзҡ„ж•°жҚ®еҸҢеҗ‘з»‘е®ҡ

е®ҡд№үжҲ–дҝ®ж”№еҶ…йғЁеұһжҖ§

var obj = {}
Object.defineProperty(obj,'hello',{
    get:function(){
        console.log('getж–№жі•иў«и°ғз”ЁдәҶ');
    }
    set:function(val){
        console.log('setж–№жі•иў«и°ғз”ЁдәҶпјҢеҸӮж•°жҳҜ' +val);
    }
})
obj.hello   //getж–№жі•иў«и°ғз”ЁдәҶ
obj.hello = 'abc'   //setж–№жі•иў«и°ғз”ЁдәҶпјҢеҸӮж•°жҳҜabc
get е’Ң setж–№жі•еҶ…йғЁthisжҢҮеҗ‘зҡ„йғҪжҳҜobj,иҝҷж„Ҹе‘ізқҖgetе’ҢsetеҸҜд»Ҙдҝ®ж”№ж“ҚдҪңеҜ№иұЎеҶ…йғЁзҡ„еҖјгҖӮ

е®һдҫӢпјҡжһҒз®ҖеҸҢеҗ‘ж•°жҚ®зҡ„е®һзҺ°

<input type="text" id="a">
<span id="b"></span>

<script>
    var obj = {}
    Object.defineProperty(obj,'hello',{
        set:function(newVal){
            document.getElementById('a').value = newVal;
            document.getElementById('b').innerHTML = newVal;
        }
    });
    document.addEventLister('keyup',function(e){
        obj.hello = e.target.value;
    })
</script>

# з®Җиҝ°Vueзҡ„е“Қеә”ејҸеҺҹзҗҶ

еҪ“дёҖдёӘVueе®һдҫӢеҲӣе»әж—¶пјҢvueдјҡйҒҚеҺҶdataйҖүйЎ№зҡ„еұһжҖ§пјҢз”Ё Object.defineProperty е°Ҷе®ғд»¬иҪ¬дёә getter/setterе№¶дё”еңЁеҶ…йғЁиҝҪиёӘзӣёе…ідҫқиө–пјҢеңЁеұһжҖ§иў«и®ҝй—®е’Ңдҝ®ж”№ж—¶йҖҡзҹҘеҸҳеҢ–гҖӮ жҜҸдёӘз»„д»¶е®һдҫӢйғҪжңүзӣёеә”зҡ„ watcher зЁӢеәҸе®һдҫӢпјҢе®ғдјҡеңЁз»„д»¶жёІжҹ“зҡ„иҝҮзЁӢдёӯжҠҠеұһжҖ§и®°еҪ•дёәдҫқиө–пјҢд№ӢеҗҺеҪ“дҫқиө–йЎ№зҡ„ setter иў«и°ғз”Ёж—¶пјҢдјҡйҖҡзҹҘ watcher йҮҚж–°и®Ўз®—пјҢд»ҺиҖҢиҮҙдҪҝе®ғе…іиҒ”зҡ„з»„д»¶еҫ—д»Ҙжӣҙж–°гҖӮ

# Vueзҡ„з”ҹе‘Ҫе‘Ёжңҹ

еңЁи°ҲеҲ°Vueзҡ„з”ҹе‘Ҫе‘Ёжңҹзҡ„ж—¶еҖҷпјҢжҲ‘д»¬йҰ–е…ҲйңҖиҰҒеҲӣе»әдёҖдёӘе®һдҫӢпјҢд№ҹе°ұжҳҜеңЁ new Vue ( ) зҡ„еҜ№иұЎиҝҮзЁӢеҪ“дёӯпјҢйҰ–е…Ҳжү§иЎҢдәҶinitпјҲinitжҳҜvueз»„д»¶йҮҢйқўй»ҳи®ӨеҺ»жү§иЎҢзҡ„пјүпјҢеңЁinitзҡ„иҝҮзЁӢеҪ“дёӯйҰ–е…Ҳи°ғз”ЁдәҶbeforeCreateпјҢз„¶еҗҺеңЁinjectionsпјҲжіЁе°„пјүе’ҢreactivityпјҲеҸҚеә”жҖ§пјүзҡ„ж—¶еҖҷпјҢе®ғдјҡеҶҚеҺ»и°ғз”ЁcreatedгҖӮжүҖд»ҘеңЁinitзҡ„ж—¶еҖҷпјҢдәӢд»¶е·Із»Ҹи°ғз”ЁдәҶпјҢжҲ‘д»¬еңЁbeforeCreateзҡ„ж—¶еҖҷеҚғдёҮдёҚиҰҒеҺ»дҝ®ж”№dataйҮҢйқўиөӢеҖјзҡ„ж•°жҚ®пјҢжңҖж—©д№ҹиҰҒж”ҫеңЁcreatedйҮҢйқўеҺ»еҒҡпјҲж·»еҠ дёҖдәӣиЎҢдёәпјүгҖӮ

еҪ“createdе®ҢжҲҗд№ӢеҗҺпјҢе®ғдјҡеҺ»еҲӨж–ӯinstanceпјҲе®һдҫӢпјүйҮҢйқўжҳҜеҗҰеҗ«жңүвҖңelвҖқoptionпјҲйҖүйЎ№пјүпјҢеҰӮжһңжІЎжңүзҡ„иҜқпјҢе®ғдјҡи°ғз”Ёvm.$mount(el)иҝҷдёӘж–№жі•пјҢз„¶еҗҺжү§иЎҢдёӢдёҖжӯҘпјӣеҰӮжһңжңүзҡ„иҜқпјҢзӣҙжҺҘжү§иЎҢдёӢдёҖжӯҘгҖӮзҙ§жҺҘзқҖдјҡеҲӨж–ӯжҳҜеҗҰеҗ«жңүвҖңtemplateвҖқиҝҷдёӘйҖүйЎ№пјҢеҰӮжһңжңүзҡ„иҜқпјҢе®ғдјҡжҠҠtemplateи§ЈжһҗжҲҗдёҖдёӘrender function пјҢиҝҷжҳҜдёҖдёӘtemplateзј–иҜ‘зҡ„иҝҮзЁӢпјҢз»“жһңжҳҜи§ЈжһҗжҲҗдәҶrenderеҮҪж•°пјҡ

render (h) { return h('div', {}, this.text) }

и§ЈйҮҠдёҖдёӢпјҢrenderеҮҪж•°йҮҢйқўзҡ„дј еҸӮhе°ұжҳҜVueйҮҢйқўзҡ„createElementж–№жі•пјҢreturnиҝ”еӣһдёҖдёӘcreateElementж–№жі•пјҢе…¶дёӯиҰҒдј 3дёӘеҸӮж•°пјҢз¬¬дёҖдёӘеҸӮж•°е°ұжҳҜеҲӣе»әзҡ„divж Үзӯҫпјӣз¬¬дәҢдёӘеҸӮж•°дј дәҶдёҖдёӘеҜ№иұЎпјҢеҜ№иұЎйҮҢйқўеҸҜд»ҘжҳҜжҲ‘д»¬з»„д»¶дёҠйқўзҡ„propsпјҢжҲ–иҖ…жҳҜдәӢд»¶д№Ӣзұ»зҡ„дёңиҘҝпјӣз¬¬дёүдёӘеҸӮж•°е°ұжҳҜdivж ҮзӯҫйҮҢйқўзҡ„еҶ…е®№пјҢиҝҷйҮҢжҲ‘д»¬жҢҮеҗ‘дәҶdataйҮҢйқўзҡ„textгҖӮ

дҪҝз”ЁrenderеҮҪж•°зҡ„з»“жһңе’ҢжҲ‘д»¬д№ӢеүҚдҪҝз”Ёtemplateи§ЈжһҗеҮәжқҘзҡ„з»“жһңжҳҜдёҖж ·зҡ„гҖӮrenderеҮҪж•°жҳҜеҸ‘з”ҹеңЁbeforeMountе’Ңmountedд№Ӣй—ҙзҡ„пјҢиҝҷд№ҹд»Һдҫ§йқўиҜҙжҳҺдәҶпјҢеңЁbeforeMountзҡ„ж—¶еҖҷпјҢ$elиҝҳеҸӘжҳҜжҲ‘д»¬еңЁHTMLйҮҢйқўеҶҷзҡ„иҠӮзӮ№пјҢз„¶еҗҺеҲ°mountedзҡ„ж—¶еҖҷпјҢе®ғе°ұжҠҠжёІжҹ“еҮәжқҘзҡ„еҶ…е®№жҢӮиҪҪеҲ°дәҶDOMиҠӮзӮ№дёҠгҖӮиҝҷдёӯй—ҙзҡ„иҝҮзЁӢе…¶е®һжҳҜжү§иЎҢдәҶrender functionзҡ„еҶ…е®№гҖӮ

еңЁдҪҝз”Ё.vueж–Үд»¶ејҖеҸ‘зҡ„иҝҮзЁӢеҪ“дёӯпјҢжҲ‘д»¬еңЁйҮҢйқўеҶҷдәҶtemplateжЁЎжқҝпјҢеңЁз»ҸиҝҮдәҶvue-loaderзҡ„еӨ„зҗҶд№ӢеҗҺпјҢе°ұеҸҳжҲҗдәҶrender functionпјҢжңҖз»Ҳж”ҫеҲ°дәҶvue-loaderи§ЈжһҗиҝҮзҡ„ж–Үд»¶йҮҢйқўгҖӮиҝҷж ·еҒҡжңүд»Җд№ҲеҘҪеӨ„е‘ўпјҹеҺҹеӣ жҳҜз”ұдәҺеңЁи§ЈжһҗtemplateеҸҳжҲҗrender functionзҡ„иҝҮзЁӢпјҢжҳҜдёҖдёӘйқһеёёиҖ—ж—¶зҡ„иҝҮзЁӢпјҢvue-loaderеё®жҲ‘д»¬еӨ„зҗҶдәҶиҝҷдәӣеҶ…е®№д№ӢеҗҺпјҢеҪ“жҲ‘д»¬еңЁйЎөйқўдёҠжү§иЎҢvueд»Јз Ғзҡ„ж—¶еҖҷпјҢж•ҲзҺҮдјҡеҸҳеҫ—жӣҙй«ҳгҖӮ

beforeMountеңЁжңүдәҶrender functionзҡ„ж—¶еҖҷжүҚдјҡжү§иЎҢпјҢеҪ“жү§иЎҢе®Ңrender functionд№ӢеҗҺпјҢе°ұдјҡи°ғз”ЁmountedиҝҷдёӘй’©еӯҗпјҢеңЁmountedжҢӮиҪҪе®ҢжҜ•д№ӢеҗҺпјҢиҝҷдёӘе®һдҫӢе°ұз®—жҳҜиө°е®ҢжөҒзЁӢдәҶгҖӮ

еҗҺз»ӯзҡ„й’©еӯҗеҮҪж•°жү§иЎҢзҡ„иҝҮзЁӢйғҪжҳҜйңҖиҰҒеӨ–йғЁзҡ„и§ҰеҸ‘жүҚдјҡжү§иЎҢгҖӮжҜ”еҰӮиҜҙжңүж•°жҚ®зҡ„еҸҳеҢ–пјҢдјҡи°ғз”ЁbeforeUpdateпјҢз„¶еҗҺз»ҸиҝҮVirtual DOMпјҢжңҖеҗҺupdatedжӣҙж–°е®ҢжҜ•гҖӮеҪ“з»„д»¶иў«й”ҖжҜҒзҡ„ж—¶еҖҷпјҢе®ғдјҡи°ғз”ЁbeforeDestoryпјҢд»ҘеҸҠdestoryedгҖӮ


beforeCreate()    //  (еҲӣе»әеүҚ) еңЁж•°жҚ®и§ӮжөӢе’ҢеҲқе§ӢеҢ–дәӢд»¶иҝҳжңӘејҖе§Ӣ
created()         //  (еҲӣе»әеҗҺ) е®ҢжҲҗж•°жҚ®и§ӮжөӢпјҢеұһжҖ§е’Ңж–№жі•зҡ„иҝҗз®—пјҢеҲқе§ӢеҢ–дәӢд»¶
beforeMount()     //  (еҠ иҪҪеүҚ) жҢӮеңЁејҖе§Ӣд№ӢеүҚиў«и°ғз”ЁпјҢж•°жҚ®иҝҳжІЎжңүжҢӮеңЁеҲ°йЎөйқўдёҠ
mounted()         //  (еҠ иҪҪеҗҺ) еңЁelиў«ж–°еҲӣе»әзҡ„vm.$elжӣҝжҚўпјҢе№¶дё”жҢӮеңЁеҲ°е®һдҫӢдёҠд№ӢеҗҺи°ғз”Ё
beforeUpdate()    //  (жӣҙж–°еүҚ) еңЁж•°жҚ®жӣҙж–°д№ӢеүҚи°ғз”Ё
updated()         //  (жӣҙж–°еҗҺ) еңЁз”ұдәҺж•°жҚ®ж”№еҠЁеҜјиҮҙзҡ„иҷҡжӢҹDOMйҮҚж–°жёІжҹ“д№ӢеҗҺи°ғз”Ё
beforeDestroy()   //  (й”ҖжҜҒеүҚ) еңЁе®һдҫӢй”ҖжҜҒд№ӢеүҚи°ғз”ЁгҖӮе®һдҫӢд»Қз„¶е®Ңе…ЁеҸҜз”ЁгҖӮ
destroyed()       //  (й”ҖжҜҒеҗҺ) еңЁе®һдҫӢй”ҖжҜҒд№ӢеҗҺи°ғз”ЁгҖӮи°ғз”ЁеҗҺпјҢжүҖжңүзҡ„дәӢд»¶зӣ‘еҗ¬еҷЁдјҡиў«з§»йҷӨгҖӮ

  • 9.д»Җд№ҲжҳҜvueзҡ„и®Ўз®—еұһжҖ§пјҹ

зӯ”пјҡеңЁжЁЎжқҝдёӯж”ҫе…ҘеӨӘеӨҡзҡ„йҖ»иҫ‘дјҡи®©жЁЎжқҝиҝҮйҮҚдё”йҡҫд»Ҙз»ҙжҠӨпјҢеңЁйңҖиҰҒеҜ№ж•°жҚ®иҝӣиЎҢеӨҚжқӮеӨ„зҗҶпјҢдё”еҸҜиғҪеӨҡж¬ЎдҪҝз”Ёзҡ„жғ…еҶөдёӢпјҢе°ҪйҮҸйҮҮеҸ–и®Ўз®—еұһжҖ§зҡ„ж–№ејҸгҖӮеҘҪеӨ„пјҡв‘ дҪҝеҫ—ж•°жҚ®еӨ„зҗҶз»“жһ„жё…жҷ°пјӣв‘Ўдҫқиө–дәҺж•°жҚ®пјҢж•°жҚ®жӣҙж–°пјҢеӨ„зҗҶз»“жһңиҮӘеҠЁжӣҙж–°пјӣв‘ўи®Ўз®—еұһжҖ§еҶ…йғЁthisжҢҮеҗ‘vmе®һдҫӢпјӣв‘ЈеңЁtemplateи°ғз”Ёж—¶пјҢзӣҙжҺҘеҶҷи®Ўз®—еұһжҖ§еҗҚеҚіеҸҜпјӣв‘Өеёёз”Ёзҡ„жҳҜgetterж–№жі•пјҢиҺ·еҸ–ж•°жҚ®пјҢд№ҹеҸҜд»ҘдҪҝз”Ёsetж–№жі•ж”№еҸҳж•°жҚ®пјӣв‘ҘзӣёиҫғдәҺmethodsпјҢдёҚз®Ўдҫқиө–зҡ„ж•°жҚ®еҸҳдёҚеҸҳпјҢmethodsйғҪдјҡйҮҚж–°и®Ўз®—пјҢдҪҶжҳҜдҫқиө–ж•°жҚ®дёҚеҸҳзҡ„ж—¶еҖҷcomputedд»Һзј“еӯҳдёӯиҺ·еҸ–пјҢдёҚдјҡйҮҚж–°и®Ўз®—гҖӮ
